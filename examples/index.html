Why use docxtemplater?

Install & Run

The best way to get docxtemplater is is via npm

npm install docxtemplater --save

Then create "input.docx" file containing "Hello {firstname} {lastname} !"

And create the following index.js file:

var fs=require('fs');
Docxtemplater = require('docxtemplater');

content = fs
    .readFileSync(__dirname+"/input.docx","binary");

doc=new Docxtemplater(content);

doc.setData({
    "firstname":"John",
    "lastname":"Doe"
});

doc.render();

var buf = doc.getZip()
             .generate({type:"nodebuffer"});

fs.writeFileSync(__dirname+"/output.docx",buf);

And that's it! docxtemplater will replace all keys passed to setData inside the input docx
